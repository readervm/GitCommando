## Setup your identity: username and email address 
git config --global user.name "Body Designer"
git config --global user.email "BD@Lex2020.com"

## Get a copy of remote repo into local directory
git clone <git URL>

## List all branches
git branch

## Check on Git log 
git log #View commit history
git log --pretty=format:"%h - %an, %ar : %s" #Specify log format
git log --all --decorate --oneline --graph   #Specify graph format

# Create a new branch
git branch <branchname>

# Move head to a branch
git checkout <branch>

# Check list of files modified / added / deleted on local repo
git status

# Add single file to staging 
git add <filename> 

# Add all files modified / added / deleted to staging 
git add .

# Commit all files staging 
git commit -m "<commit msg>"

# Add and commit all files in one command
git commit -a -m "<commit msg>"

# Push code to remote Github repository
## Use this when <branchname> is not exists in remote but local(one time cmd)
git push -u origin <branchname>

## Use this when <branchname> is already exists
git push
